class AudioRecorder {
  async start(): Promise<void> {
    // Only Android needs explicit recording mode setup
    if (Platform.OS === 'android') {
      await Audio.setAudioModeAsync({
        staysActiveInBackground: true,
        shouldDuckAndroid: false,       // Don't duck other audio while recording
      }).catch(err => {
        console.warn('[AudioRecorder] Android recording mode setup failed:', err);
      });
    }
    // iOS: System handles recording mode, don't touch it
    
    // ... start recording (same for both platforms)
  }

  async stop(): Promise<URI | Blob> {
    // Only Android needs to revert mode
    if (Platform.OS === 'android') {
      await Audio.setAudioModeAsync({
        staysActiveInBackground: true,
        shouldDuckAndroid: true,        // Re-enable ducking
      }).catch(err => {
        console.warn('[AudioRecorder] Android playback mode revert failed:', err);
      });
    }
    // iOS: Already in correct state
    
    // ... stop recording (same for both)
  }
}
Why safe for iOS:

✅ iOS audio mode is never touched during recording
✅ iOS uses system defaults throughout
✅ No risk of breaking iOS recording/playback
